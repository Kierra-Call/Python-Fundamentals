# Data Types
1. String -> " " OR ' '
2. Number -> Int AND Float
3. Bool -> True or False

    NUMBERS:
        0 -> False
        1, 2, 3, etc... -> True.
    Strings:
        "" -> False
        " " -> True
        "Tyler" -> True
